package fes.aragon.compilador;

import java_cup.runtime.*;
import java.io.*;
import java.util.ArrayList;
import fes.aragon.tablerointerprete.modelo.Comando; // Asegúrate de que esta ruta coincida con donde creaste la clase Comando

// Código que se añade a la clase parser generada
parser code {:
    public Lexer lex;

    // ¡Aquí está nuestra lista para el Camino B!
    public ArrayList<Comando> listaComandos = new ArrayList<>();

    // Constructor opcional para pasar el lexer fácilmente
    public parser(Lexer lex) {
        super(lex);
        this.lex = lex;
    }

    // Método para manejar errores sintácticos
    public void syntax_error(Symbol st) {
        throw new RuntimeException("Error sintáctico: Componente <" + st.value + "> no esperado. Línea " +
        (lex.getYyline() + 1) + ", Columna " + (lex.getYycolumn() + 1));
    }

    // Método para errores irrecuperables
    public void unrecovered_syntax_error(Symbol st) throws java.lang.Exception {
        System.out.println("Error sintáctico irrecuperable en la línea " +
                           (lex.getYyline() + 1) + ", Columna " + (lex.getYycolumn() + 1));
    }
:};

/* Terminales */
terminal INICIO, DERECHA, IZQUIERDA, ABAJO, ARRIBA, MOVER, REPITE, PARAR;
terminal Integer NUMERO;

/* No terminales */
non terminal programa, lista_instrucciones, instruccion;

start with programa;

/* --- Gramática --- */

programa ::= lista_instrucciones:li
    {:
        // Le indicamos explícitamente a Java que 'li' es un ArrayList
        parser.listaComandos = (ArrayList<Comando>) li;
        RESULT = li;
    :};

lista_instrucciones ::= lista_instrucciones:li instruccion:i
    {:
        // Hacemos el cast antes de usar addAll
        ((ArrayList<Comando>) li).addAll((ArrayList<Comando>) i);
        RESULT = li;
    :}
    | instruccion:i
    {:
        RESULT = i;
    :};

instruccion ::= INICIO NUMERO:x NUMERO:y
    {:
        ArrayList<Comando> l = new ArrayList<>();
        l.add(new Comando("inicioX", x));
        l.add(new Comando("inicioY", y));
        RESULT = l;
    :}
    | DERECHA
    {:
        ArrayList<Comando> l = new ArrayList<>(); l.add(new Comando("derecha")); RESULT = l;
    :}
    | IZQUIERDA
    {:
        ArrayList<Comando> l = new ArrayList<>(); l.add(new Comando("izquierda")); RESULT = l;
    :}
    | ABAJO
    {:
        ArrayList<Comando> l = new ArrayList<>(); l.add(new Comando("abajo")); RESULT = l;
    :}
    | ARRIBA
    {:
        ArrayList<Comando> l = new ArrayList<>(); l.add(new Comando("arriba")); RESULT = l;
    :}
    | MOVER NUMERO:n
    {:
        ArrayList<Comando> l = new ArrayList<>(); l.add(new Comando("mover", n)); RESULT = l;
    :}
    | REPITE NUMERO:n lista_instrucciones:li PARAR
    {:
        ArrayList<Comando> l = new ArrayList<>();
        // ¡LA MAGIA DEL COMPILADOR! Multiplicamos el bloque interno 'n' veces
        for(int i = 0; i < n; i++){
            // Agregamos el (ArrayList<Comando>) justo antes de li
            l.addAll((ArrayList<Comando>) li);
        }
        RESULT = l;
    :}
    | error
    {:
        RESULT = new ArrayList<>();
    :};
